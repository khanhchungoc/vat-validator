```jsx
export default function ProgressBar({ invoices }) {
  if (invoices.length === 0) return null

  const done = invoices.filter(i => !['pending', 'processing'].includes(i.status)).length
  const total = invoices.length
  const pct = total > 0 ? Math.round((done / total) * 100) : 0

  const passed = invoices.filter(i => i.status === 'pass').length
  const failed = invoices.filter(i => ['invalid-invoice', 'invalid-business'].includes(i.status)).length
  const skipped = invoices.filter(i => i.status === 'skipped').length

  return (
    <div className="progress-container">
      <div className="progress-header">
        <span className="progress-label">{done} / {total} invoices processed</span>
        <span className="progress-pct">{pct}%</span>
      </div>
      <div className="progress-track">
        <div className="progress-fill" style={{ width: `${pct}%` }} />
      </div>
      <div className="progress-stats">
        <span style={{ color: 'var(--pass)' }}>✅ {passed} Pass</span>
        <span style={{ color: 'var(--fail)' }}>❌ {failed} Failed</span>
        <span style={{ color: 'var(--skip)' }}>⚠️ {skipped} Skipped</span>
      </div>
    </div>
  )
}
```

          ))}

          <button className="btn-primary" onClick={onProceed}>Proceed Anyway</button>
        </div>
      </div>

  )
}

- [ ] **Step 2: Wire everything together in `src/App.jsx`** — final full version

```jsx
import { useState } from 'react'
import { useWebSocket } from './useWebSocket'
import DropZone from './components/DropZone'
import ManualEntryForm from './components/ManualEntryForm'
import InvoiceQueue from './components/InvoiceQueue'
import CaptchaModal from './components/CaptchaModal'
import ProgressBar from './components/ProgressBar'
import ModeToggle from './components/ModeToggle'
import DuplicateWarning from './components/DuplicateWarning'
import ErrorBanner from './components/ErrorBanner'
import StepButton from './components/StepButton'
import ResumePanel from './components/ResumePanel'
import DownloadButtons from './components/DownloadButtons'

export default function App() {
  const [invoices, setInvoices] = useState([])
  const [showManualForm, setShowManualForm] = useState(false)
  const [captcha, setCaptcha] = useState(null)
  const [stepWaiting, setStepWaiting] = useState(false)
  const [mode, setMode] = useState('auto')
  const [isProcessing, setIsProcessing] = useState(false)
  const [duplicates, setDuplicates] = useState([])
  const [processingError, setProcessingError] = useState(null)
  const [downloadUrls, setDownloadUrls] = useState({ pdfUrl: null, xlsxUrl: null })
  const [currentSessionDir, setCurrentSessionDir] = useState(null)
  const [wsStatus, setWsStatus] = useState('Connecting...')

  const { send } = useWebSocket((msg) => {
    switch (msg.type) {
      case 'pong': setWsStatus('Connected ✅'); break
      case 'invoice-added': setInvoices(p => [...p, msg.payload]); break
      case 'captcha-required': setCaptcha(msg.payload); break
      case 'invoice-status':
        setInvoices(p => p.map(i => i.id === msg.payload.id ? { ...i, ...msg.payload } : i))
        setCaptcha(null)
        setProcessingError(null)
        break
      case 'step-waiting': setStepWaiting(true); break
      case 'mode-changed': setMode(msg.payload); break
      case 'processing-error': setProcessingError(msg.payload.message); break
      case 'batch-complete':
        setIsProcessing(false); setStepWaiting(false)
        if (msg.payload?.pdfUrl) setDownloadUrls(msg.payload)
        break
      case 'error': alert(msg.payload); break
    }
  })

  function handleFilesUploaded(results) {
    const added = results.filter(r => r.ok).map(r => r.invoice)
    const dupes = results.filter(r => !r.ok && r.error?.includes('Duplicate')).map(r => r.error)
    setInvoices(p => [...p, ...added])
    if (dupes.length > 0) setDuplicates(dupes)
  }

  function handleManualSubmit(data) {
    send({ type: 'add-manual-invoice', payload: data })
    setShowManualForm(false)
  }

  async function handleStartProcessing() {
    let sessionDir = currentSessionDir
    if (!sessionDir) {
      const res = await fetch('http://localhost:3001/sessions/new', { method: 'POST' })
      const data = await res.json()
      sessionDir = data.sessionDir
      setCurrentSessionDir(sessionDir)
    }
    setIsProcessing(true)
    send({ type: 'start-processing', payload: { sessionDir, mode } })
  }

  async function handleResume(sessionDir) {
    const res = await fetch('http://localhost:3001/sessions/resume', {
      method: 'POST', headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ sessionDir })
    })
    const data = await res.json()
    setInvoices(data.invoices)
    setCurrentSessionDir(sessionDir)
  }

  function handleModeChange(newMode) {
    setMode(newMode)
    if (isProcessing) send({ type: 'set-mode', payload: newMode })
  }

  return (
    <div className="app">
      <header className="app-header">
        <h1>VATOCR</h1>
        <span className="ws-status">{wsStatus}</span>
      </header>
      <main className="app-main">
        <ResumePanel onResume={handleResume} />
        <DropZone onFilesUploaded={handleFilesUploaded} />
        <button className="btn-secondary" style={{ marginTop: 12 }}
          onClick={() => setShowManualForm(true)}>
          + Add Invoice Manually
        </button>
        {invoices.length > 0 && !isProcessing && (
          <div style={{ marginTop: 20, display: 'flex', gap: 12, alignItems: 'center' }}>
            <ModeToggle mode={mode} onChange={handleModeChange} disabled={false} />
            <button className="btn-primary" onClick={handleStartProcessing}>
              ▶ Start Processing ({invoices.filter(i => i.status === 'pending').length} pending)
            </button>
          </div>
        )}
        {isProcessing && (
          <ModeToggle mode={mode} onChange={handleModeChange} disabled={false} />
        )}
        <ErrorBanner
          error={processingError}
          onRetry={() => { send({ type: 'captcha-answer', payload: { answer: 'retry' } }); setProcessingError(null) }}
          onSkip={() => { send({ type: 'skip-invoice' }); setProcessingError(null) }}
        />
        <ProgressBar invoices={invoices} />
        <StepButton visible={stepWaiting} onStep={() => {
          send({ type: 'advance-step' }); setStepWaiting(false)
        }} />
        <InvoiceQueue invoices={invoices} />
        <DownloadButtons pdfUrl={downloadUrls.pdfUrl} xlsxUrl={downloadUrls.xlsxUrl} />
      </main>
      {showManualForm && (
        <ManualEntryForm onSubmit={handleManualSubmit} onClose={() => setShowManualForm(false)} />
      )}
      {captcha && (
        <CaptchaModal
          imageBase64={captcha.image}
          attempt={captcha.attempt}
          onSubmit={(answer) => { send({ type: 'captcha-answer', payload: { answer } }); setCaptcha(null) }}
          onSkip={() => { send({ type: 'skip-invoice' }); setCaptcha(null) }}
        />
      )}
      {duplicates.length > 0 && (
        <DuplicateWarning
          duplicates={duplicates}
          onRemove={() => setDuplicates([])}
          onProceed={() => setDuplicates([])}
        />
      )}
    </div>
  )
}
```
